\begin{beispiel}[K7 - KKK]{1} %PUNKTE DES BEISPIELS
			Gib eine Parameterdarstellung der Ellipse mit der Gleichung $x²+4y²=100$ an!
			
			\antwort{$\frac{x²}{100}+\frac{y²}{25}=1 \Rightarrow a=10$ und $b=5 \Rightarrow \binom{x(t)}{y(t)}=\binom{10\cdot\cos t}{5\cdot\sin t}$ mit $t\in[0;2\pi]$}
			\end{beispiel}
