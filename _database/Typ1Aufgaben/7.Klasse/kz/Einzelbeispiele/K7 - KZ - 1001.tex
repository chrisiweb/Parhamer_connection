\section{K7 - KZ - 1001 - Definition von i - ZO - Lösungswege 7. Klasse}

\begin{beispiel}[K7 - KZ]{1}
Gegeben sind Aussagen über komplexe Zahlen. Ordne so zu, dass wahre Aussagen entstehen!

\zuordnen{
				R1={$i^n=i$},				% Response 1
				R2={$i^n=1$},				% Response 2
				R3={$i^n=-1$},				% Response 3
				R4={$i^n=-i$},				% Response 4
				%% Moegliche Zuordnungen: %%
				A={für $n\in\mathbb{N}$}, 				%Moeglichkeit A  
				B={für $n=4k$ und $k\in\mathbb{N}$}, 				%Moeglichkeit B  
				C={für $n=4k+1$ und $k\in\mathbb{N}$}, 				%Moeglichkeit C  
				D={für $n=4k+2$ und $k\in\mathbb{N}$}, 				%Moeglichkeit D  
				E={für $n=4k+3$ und $k\in\mathbb{N}$}, 				%Moeglichkeit E  
				F={für $n\in\mathbb{N}^*$}, 				%Moeglichkeit F  
				%% LOESUNG: %%
				A1={C},				% 1. richtige Zuordnung
				A2={B},				% 2. richtige Zuordnung
				A3={D},				% 3. richtige Zuordnung
				A4={E},				% 4. richtige Zuordnung
				}
\end{beispiel}
