				\zuordnen{
								R1={$\{x\in\mathbb{N}_u\mid 5\leq x\leq 15\}$},				% Response 1
								R2={$\{x\in\mathbb{N}\mid 2<x\leq 12\}$},				% Response 2
								R3={$\{x\in\mathbb{Z}\mid -5\leq x<2\}$},				% Response 3

								A={$\{-5; -4; -3; -2; -1; 0; 1\}$}, 				%Moeglichkeit A  
								B={$\{\}$}, 				%Moeglichkeit B

								F={$\{7;9;11;13;15\}$}, 				%Moeglichkeit F

								}
								\end{scriptsize}
