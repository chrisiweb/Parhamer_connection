Gegeben seien folgende Aussagen:
				
				A: 5 ist Teiler von 15.\\
				B: 7 ist eine ungerade Zahl.\\
				C: 9 ist eine Primzahl.\\
				D: 8 ist durch 2 teilbar.\\
				E: 8 ist eine ungerade Zahl.\\
				F: 2 ist eine Primzahl.
				
				Kreuze jene Verknüpfung(en) von Aussagen an, die wahr sind!
				
				\multiplechoice[5]{  %Anzahl der Antwortmoeglichkeiten, Standard: 5
								L1={$A\wedge B$},   %1. Antwortmoeglichkeit 
								L2={$A\wedge C$},   %2. Antwortmoeglichkeit
								L3={$A\vee D$},   %3. Antwortmoeglichkeit
								L4={$E\Rightarrow F$},   %4. Antwortmoeglichkeit
								L5={$E\vee C$},	 %5. Antwortmoeglichkeit
								L6={$B\Leftrightarrow F$},	 %6. Antwortmoeglichkeit
								L7={$D\Rightarrow E$},	 %7. Antwortmoeglichkeit
								L8={$C\Leftrightarrow E$},	 %8. Antwortmoeglichkeit
								L9={},	 %9. Antwortmoeglichkeit
								%% LOESUNG: %%
								A1=1,  % 1. Antwort
								A2=3,	 % 2. Antwort
								A3=4,  % 3. Antwort
								A4=6,  % 4. Antwort
								A5=0,  % 5. Antwort
								}
