Die nachfolgenden Aussagen beziehen sich auf die Lösung einer Differentialgleichung.

\lueckentext{
				text={Die Funktion $f$ mit \gap ist eine Lösung der Differentialgleichung \gap.}, 	%Lueckentext Luecke=\gap
				L1={$f(x)=\sin(x)$}, 		%1.Moeglichkeit links  
				L2={$f(x)=x^2$}, 		%2.Moeglichkeit links
				L3={$f(x)=2^x$}, 		%3.Moeglichkeit links
				R1={$f'(x)+f(x)=0$}, 		%1.Moeglichkeit rechts 
				R2={$f'(x)=2\cdot f(x)$}, 		%2.Moeglichkeit rechts
				R3={$f'(x)\cdot f(x)=2x^3$}, 		%3.Moeglichkeit rechts
				%% LOESUNG: %%
				A1=2,   % Antwort links
				A2=3		% Antwort rechts 
				}
