\section{WS 4.1 - 2 Wähleranteil - OA - BIFIE}

\begin{beispiel}[WS 4.1]{1} %PUNKTE DES BEISPIELS
Bei einer Stichprobe von $n = 500$ Personen gaben 120 Personen an, sie würden die Partei $A$
wählen. 	\leer

Gib das 95-\%-Konfidenzintervall $KI$ für den Wähleranteil der Partei $A$ an. \leer

$KI=\rule{5cm}{0.3pt}$	

\antwort{
$KI=[0,203;0277]$ bzw. $KI=0,24\mp 0,037$ \leer

Lösungsintervall für die untere Grenze: $[0,20; 0,21]$\\
Lösungsintervall für die obere Grenze: $[0,27; 0,28]$ }
\end{beispiel}
