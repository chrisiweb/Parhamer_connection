\section{AN 3.3 - 17 Charakteristika einer Polynomfunktion - LT - BIFIE}

\begin{beispiel}[AN 3.3]{1} %PUNKTE DES BEISPIELS
				Von einer Polynomfunktion $f$ ist Folgendes bekannt: $f(2)=0,f'(2)=0$ und $f''(2)=1$.
				
				\lueckentext{
								text={$f$ hat an der Stelle \gap sicher \gap.}, 	%Lueckentext Luecke=\gap
								L1={$x=0$}, 		%1.Moeglichkeit links  
								L2={$x=1$}, 		%2.Moeglichkeit links
								L3={$x=2$}, 		%3.Moeglichkeit links
								R1={ein lokales Minimum}, 		%1.Moeglichkeit rechts 
								R2={ein lokales Maximum}, 		%2.Moeglichkeit rechts
								R3={eine Wendestelle}, 		%3.Moeglichkeit rechts
								%% LOESUNG: %%
								A1=3,   % Antwort links
								A2=1		% Antwort rechts 
								}
\end{beispiel}
