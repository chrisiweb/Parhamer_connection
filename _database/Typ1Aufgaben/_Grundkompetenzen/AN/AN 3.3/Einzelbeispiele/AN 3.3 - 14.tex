								L4={$f'(2)=0$},   %4. Antwortmoeglichkeit
								L5={$f''(2)=0$},	 %5. Antwortmoeglichkeit
								L6={},	 %6. Antwortmoeglichkeit
								L7={},	 %7. Antwortmoeglichkeit
								L8={},	 %8. Antwortmoeglichkeit
								L9={},	 %9. Antwortmoeglichkeit
								%% LOESUNG: %%
								A1=1,  % 1. Antwort
								A2=3,	 % 2. Antwort

								}
