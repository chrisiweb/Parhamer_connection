\section{AN 2.1 - 2 Ableitung von Sinus- und Cosinus-Funktion - ZO - BIFIE}

\begin{beispiel}[AN 2.1]{1} %PUNKTE DES BEISPIELS
				Gegeben sind vier Funktionen und sechs Ableitungsfunktionen.

Ordne den Funktionen die richtige Ableitungsfunktion $f'$ zu!

\zuordnen{
				title1={Ableitungsfunktionen}, 		%Titel Antwortmoeglichkeiten
				A={\small{$f'(x)=-\cos(x)+2\cdot\sin(x)$}}, 				%Moeglichkeit A  
				B={$f'(x)=2\cdot\cos(x)+\sin(x)$}, 				%Moeglichkeit B  
				C={$f'(x)=2\cdot\cos(x)-\sin(x)$}, 				%Moeglichkeit C  
				D={\small{$f'(x)=-\cos(x)-2\cdot\sin(x)$}}, 				%Moeglichkeit D  
				E={$f'(x)=\cos(x)-2\cdot\sin(x)$}, 				%Moeglichkeit E  
				F={$f'(x)=2\cdot\sin(x)+\cos(x)$}, 				%Moeglichkeit F  
				title2={Funktionen},		%Titel Zuordnung
				R1={$f(x)=2\cdot\cos(x)-\sin(x)$},				%1. Antwort rechts
				R2={$f(x)=\cos(x)+2\cdot\sin(x)$},				%2. Antwort rechts
				R3={$f(x)=-2\cdot\cos(x)-\sin(x)$},				%3. Antwort rechts
				R4={\small{$f(x)=-\cos(x)+2\cdot\sin(x)$}},				%4. Antwort rechts
				%% LOESUNG: %%
				A1={D},				% 1. richtige Zuordnung
				A2={C},				% 2. richtige Zuordnung
				A3={A},				% 3. richtige Zuordnung
				A4={B},				% 4. richtige Zuordnung
				}
\end{beispiel}
