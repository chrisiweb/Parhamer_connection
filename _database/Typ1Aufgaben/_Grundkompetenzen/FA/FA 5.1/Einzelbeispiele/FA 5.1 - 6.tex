\begin{beispiel}[FA 5.1]{1}
Gegeben sind Exponentialfunktionen der Form $f\!:x \mapsto c\cdot a^x$. Ordne den Funktionsgleichungen bzw. den Bedingungen für die Parameter $a$ und $c$ die richtige Aussage bzw. Tabelle (aus A bis F) zu.

\zuordnen[-0.16]{
				R1={$c<0, 0<a<1$},				% Response 1
				R2={$f(x)=2 \cdot 4^x$},				% Response 2
				R3={$f(x)=3 \cdot 0,5^x$},				% Response 3
				R4={$c<0, a>1$},				% Response 4
				%% Moegliche Zuordnungen: %%
				A={Der Wert an der Stelle $x=0$ des zugehörigen exponentiellen Wachstumsprozesses liegt bei $2$.}, 				%Moeglichkeit A  
				B={Die Funktion $f$ hat an der Stelle $x=0$ einen negativen Funktionswert und ist streng monoton steigend.}, 				%Moeglichkeit B  
				C={\begin{tabular}{c|c|c|c|c}
				$x$&$0$&$1$&$2$&$3$\\ \hline
				$f(x)$&$3$&$4,5$&$6,25$&$10,13$\\
				\end{tabular}}, 				%Moeglichkeit C  
				D={Die Funktion $f$ hat an der Stelle $x=0$ einen negativen Funktionswert und ist streng monoton fallend.}, 				%Moeglichkeit D  
				E={Der Wert an der Stelle $x=0$ des zugehörigen exponentiellen Wachstumsprozesses liegt bei $4$.}, 				%Moeglichkeit E  
				F={\begin{tabular}{c|c|c|c|c}
				$x$&$0$&$1$&$2$&$3$\\ \hline
				$f(x)$&$3$&$1,5$&$0,75$&$0,375$\\
				\end{tabular}}, 				%Moeglichkeit F  
				%% LOESUNG: %%
				A1={B},				% 1. richtige Zuordnung
				A2={A},				% 2. richtige Zuordnung
				A3={F},				% 3. richtige Zuordnung
				A4={D},				% 4. richtige Zuordnung
				}
\end{beispiel}
