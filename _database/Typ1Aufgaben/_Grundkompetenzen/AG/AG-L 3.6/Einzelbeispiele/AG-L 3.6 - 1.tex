\begin{beispiel}[AG-L 3.6]{1}
Berechne den Winkel zwischen den beiden gegebenen Vektoren.

$$\vec{a}=\Vek{8}{-6}{}\text{ und }\vec{b}=\Vek{3}{5}{}$$

\antwort{$\cos(\varphi)=\dfrac{\Vek{8}{-6}{}\cdot\Vek{3}{5}{}}{\sqrt{100}\cdot\sqrt{34}}=\dfrac{-6}{\sqrt{3400}}$\quad$\Rightarrow$\quad$\varphi=\cos^{-1}\left(\dfrac{-6}{\sqrt{3400}}\right)=95,91^\circ$}
\end{beispiel}
