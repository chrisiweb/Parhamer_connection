\section{AG 3.3 - 24 - Normalvektoren finden  - ZO - VikRaf}

\begin{beispiel}[AG 3.3]{1}

Ordne den angegebenen Vektoren $\vec{a}, \vec{b}, \vec{c}, \vec{d}$ einen passenden Normalvektor $\vec{n}$ (aus A bis F) zu.

\zuordnen{
				R1={$\vec{a}=\Vek{15}{25}{}$},				% Response 1
				R2={$\vec{b}= \Vek{8x}{-12}{} $},				% Response 2
				R3={$\vec{c}= \Vek{9}{-15}{} $},				% Response 3
				R4={$\vec{d}=\Vek{18x}{12}{}$},				% Response 4
				%% Moegliche Zuordnungen: %%
				A={$\vec{n_{1}}=\Vek{-2}{3x}{}$}, 				%Moeglichkeit A  
				B={$\vec{n_{2}}=\Vek{2x}{-3}{}$}, 				%Moeglichkeit B  
				C={$\vec{n_{3}}=\Vek{-5}{3}{}$}, 				%Moeglichkeit C  
				D={$\vec{n_{4}}=\Vek{3}{-5}{}$}, 				%Moeglichkeit D  
				E={$\vec{n_{5}}=\Vek{3}{2x}{}$}, 				%Moeglichkeit E  
				F={$\vec{n_{6}}=\Vek{5}{3}{}$}, 				%Moeglichkeit F  
				%% LOESUNG: %%
				A1={C},				% 1. richtige Zuordnung
				A2={E},				% 2. richtige Zuordnung
				A3={F},				% 3. richtige Zuordnung
				A4={A},				% 4. richtige Zuordnung
				}

\end{beispiel}
