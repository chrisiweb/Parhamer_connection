\begin{beispiel}[AG 2.3]{1} %PUNKTE DES BEISPIELS
	
				Gegeben sind vier Lösungsmengen und sechs quadratische Gleichungen. Ordne jeder Lösungsmenge $L$ die entsprechende quadratische Gleichung zu!
\zuordnen{
				R1={$L=\{\, \}$},				% Response 1
				R2={$L=\{-3;3\}$},				% Response 2
				R3={$L=\{0;3\}$},				% Response 3
				R4={$L=\{3\}$},				% Response 4
				%% Moegliche Zuordnungen: %%
				A={$(x+3)^2=0$}, 				%Moeglichkeit A  
				B={$(x-3)^2=16$}, 				%Moeglichkeit B  
				C={$x\cdot(x-3)=0$}, 				%Moeglichkeit C  
				D={$-x^2=9$}, 				%Moeglichkeit D  
				E={$x^2-9=0$}, 				%Moeglichkeit E  
				F={$x^2-6x+9=0$}, 				%Moeglichkeit F  
				%% LOESUNG: %%
				A1={D},				% 1. richtige Zuordnung
				A2={E},				% 2. richtige Zuordnung
				A3={C},				% 3. richtige Zuordnung
				A4={B},				% 4. richtige Zuordnung
				}
\end{beispiel}
