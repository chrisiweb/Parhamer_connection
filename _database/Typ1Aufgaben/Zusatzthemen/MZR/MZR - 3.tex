\section{MZR - 3 - K5 - Verknüpfung von Mengen - ZO - MatKon}

\begin{beispiel}[MZR]{1}
Gegeben sind die Mengen:\\ 
$A=\{2, 4, 6, 8, 10\}$, $B=\{2, 3, 5, 7, 11\}, C=\{1, 2, 3, 5, 8, 11\}$.

Ordne den Verknüpfungen auf der linken Seite die jeweils richtige Menge auf der rechten Seite zu!

\zuordnen{
				R1={$A\backslash C$},				% Response 1
				R2={$(A\cap B)\backslash C$},				% Response 2
				R3={$(B\cup C)\cap A$},				% Response 3
				R4={$B\backslash C$},				% Response 4
				%% Moegliche Zuordnungen: %%
				A={$\{\}$}, 				%Moeglichkeit A  
				B={$\{7\}$}, 				%Moeglichkeit B  
				C={$\{7, 8, 11\}$}, 				%Moeglichkeit C  
				D={$\{4, 6, 10\}$}, 				%Moeglichkeit D  
				E={$\{1\}$}, 				%Moeglichkeit E  
				F={$\{2, 8\}$}, 				%Moeglichkeit F  
				%% LOESUNG: %%
				A1={D},				% 1. richtige Zuordnung
				A2={A},				% 2. richtige Zuordnung
				A3={F},				% 3. richtige Zuordnung
				A4={B},				% 4. richtige Zuordnung
				}
\end{beispiel}
